198 double get_value(const CatchmentAggrDataSelector& selector, data_access::ReSampleMethod m) override
199 {
200 std::string output_name = selector.get_variable_name();
201 time_t init_time = selector.get_init_time();
202 const long duration_s = selector.get_duration_secs();
203 const std::string output_units = selector.get_output_units();
204
205 // Balk if not in this instance's collection of outputs
206 if (find(providedOutputs.begin(), providedOutputs.end(), output_name) == providedOutputs.end()) {
207 std::string throw_msg; throw_msg.assign("Unknown output " + output_name + " requested from wrapped provider");
208 LOG(throw_msg, LogLevel::WARNING);
209 throw std::runtime_error(throw_msg);
210 }
211 // Balk if this instance is not ready
212 if (!isReadyToProvideData()) {
213 std::string throw_msg; throw_msg.assign("Cannot get value for " + output_name
214 + " from optional wrapped provider before it is ready");
215 LOG(throw_msg, LogLevel::WARNING);
216 throw std::runtime_error(throw_msg);
217 }
218
219 // Remember: when the instance is ready to provide data, one or both of these **must** be true:
220 // * the desired output has a default value available
221 // * the desired output is provided by the backing wrapped provider
222
223 // If this output is not available from a wrapped provider OR we should override, then return the default
224 // Remember: isDefaultOverride() is not the same as "should use default" (see function docstring)
225 if (!isSuppliedByWrappedProvider(output_name) || isDefaultOverride(output_name)) {
226 // Take note if/how often default gets used in case that information is needed later
227 recordUsingDefault(output_name);
228 return defaultValues.at(output_name);
229 }
230 // Otherwise (i.e., available from wrapped provider and no override), get from backing wrapped provider
231 else {
232 return wrapped_provider->get_value(selector,m);
233 }
234 }

258 bool isDefaultOverride(const std::string &output_name) {
259 // First, there must be a default, and there must be something to override
260 if (!isSuppliedWithDefault(output_name) || !isSuppliedByWrappedProvider(output_name)) {
261 return false;
262 }
263 // Must have positive wait indicator value
264 const auto &wait_it = defaultUsageWaits.find(output_name);
265 return wait_it != defaultUsageWaits.end() && wait_it->second > 0;
266 }

320 bool setWrappedProvider(
321 GenericDataProvider *provider) override {
322 // Disallow re-setting the provider if already validly set
323 // TODO: consider if this is still appropriate, given backing providers can be valid without providing
324 // everything (because of defaults), and so a new provider may supply more or less of what is needed
325 if (isWrappedProviderSet()) {
326 setMessage = "Cannot change wrapped provider after a valid provide has already been set";
327 return false;
328 }
329
330 // Confirm this will provide anything
331 if (provider == nullptr) {
332 setMessage = "Cannot set provider as null, as this cannot provide the values this type must proxy";
333 return false;
334 }
335
336 // Check this provides everything needed, accounting for defaults (and also tallying the outputs provided)
337 unsigned short providedByProviderCount = 0;
338 for (const std::string &requiredName : providedOutputs) {
339 // If supplied by the provider, increment our count and continue to the next required output name
340 if (isSuppliedByProvider(requiredName, provider)) {
341 ++providedByProviderCount;
342 continue;
343 }
344 // If something isn't supplied by the provider (i.e., the "else"), there has to be a default, or we bail
345 else if (!isSuppliedWithDefault(requiredName)) {
346 setMessage = "Given provider does not provide the required " + requiredName;
347 return false;
348 }
349 }
350
351 // Also make sure it provides at least 1 relevant output (possible no outputs were "needed" due to defaults)
352 if (providedByProviderCount == 0) {
353 setMessage = "Given provider does not provide any required outputs (all satisfied via defaults)";
354 return false;
355 }
356
357 // If this is good, set things and return true
358 wrapped_provider = provider;
359 setMessage.clear();
360 return true;
361 }

382 virtual void recordUsingDefault(const std::string &output_name) {
383 // Don't bother doing anything if there aren't waits assigned for this output
384 // Also, in this implementation, don't count usages until there is a backing provider that can provide this
385 auto waits_it = defaultUsageWaits.find(output_name);
386 if (waits_it != defaultUsageWaits.end() && isSuppliedByWrappedProvider(output_name)) {
387 // If this was not the last required default usage, simply decrement the wait count
388 if (waits_it->second > 1) {
389 waits_it->second -= 1;
390 }
391 // If this was the last required default usage wait, clear this entry from the collection
392 // Also, right now we are only supporting positive counts, so remove anything else as well
393 else {
394 defaultUsageWaits.erase(waits_it);
395 }
396 }
