83 size_t get_ts_index_for_time(const time_t &epoch_time) const override {
84 if (epoch_time < start_date_time_epoch) {
85 std::string throw_msg; throw_msg.assign("Forcing had bad pre-start time for index query: " + std::to_string(epoch_time));
86 LOG(throw_msg, LogLevel::WARNING);
87 throw std::out_of_range(throw_msg);
88 }
89 size_t i = 0;
90 // 1 hour
91 time_t seconds_in_time_step = 3600;
92 time_t time = start_date_time_epoch;
93 while (epoch_time >= time + seconds_in_time_step && time < end_date_time_epoch) {
94 i++;
95 time += seconds_in_time_step;
96 }
97 // The end_date_time_epoch is the epoch value of the BEGINNING of the last time step, not its end.
98 // I.e., to make sure we cover it, we have to go another time step beyond.
99 if (time >= end_date_time_epoch + 3600) {
100 std::string throw_msg; throw_msg.assign("Forcing had bad beyond-end time for index query: " + std::to_string(epoch_time));
101 LOG(throw_msg, LogLevel::WARNING);
102 throw std::out_of_range(throw_msg);
103 }
104 else {
105 return i;
106 }

119 double get_value(const CatchmentAggrDataSelector& selector, data_access::ReSampleMethod m) override
120 {
121 size_t current_index;
122 long time_remaining = selector.get_duration_secs();
123 auto init_time = selector.get_init_time();
124 auto output_name = selector.get_variable_name();
125 auto output_units = selector.get_output_units();
126
127 try {
128 current_index = get_ts_index_for_time(init_time);
129 }
130 catch (const std::out_of_range &e) {
131 std::string throw_msg; throw_msg.assign("Forcing had bad init_time " + std::to_string(init_time) + " for value request");
132 LOG(throw_msg, LogLevel::WARNING);
133 throw std::out_of_range(throw_msg);
134 }
135
136 std::vector<double> involved_time_step_values;
137
138 std::vector<long> involved_time_step_seconds;
139 long ts_involved_s;
140
141 time_t first_time_step_start_epoch = start_date_time_epoch + (current_index * 3600);
142 // Handle the first time step differently, since we need to do more to figure out how many seconds came from it
143 // Total time step size minus the offset of the beginning, before the init time
144 ts_involved_s = 3600 - (init_time - first_time_step_start_epoch);
145
146 involved_time_step_seconds.push_back(ts_involved_s);
147 involved_time_step_values.push_back(get_value_for_param_name(output_name, current_index));
148 time_remaining -= ts_involved_s;
149 current_index++;
150
151 while (time_remaining > 0) {
152 if(current_index >= time_epoch_vector.size())
153 return involved_time_step_values[involved_time_step_values.size()-1]; //TODO: Is this the right answer? Is returning any value off the end of the range valid?
154 ts_involved_s = time_remaining > 3600 ? 3600 : time_remaining;
155 involved_time_step_seconds.push_back(ts_involved_s);
156 involved_time_step_values.push_back(get_value_for_param_name(output_name, current_index));
157 time_remaining -= ts_involved_s;
158 current_index++;
159
160 }
161 double value = 0;
162 for (size_t i = 0; i < involved_time_step_values.size(); ++i) {
163 if (is_param_sum_over_time_step(output_name))
164 value += involved_time_step_values[i] * ((double)involved_time_step_seconds[i] / 3600.0);
165 else
166 value += involved_time_step_values[i] * ((double)involved_time_step_seconds[i] / (double)selector.get_duration_secs());
167 }
168
169 // Convert units
170 try {
171 return UnitsHelper::get_converted_value(available_forcings_units[output_name], value, output_units);
172 }
173 catch (const std::runtime_error& e) {
174 data_access::unit_conversion_exception uce(e.what());
175 uce.provider_model_name = "CsvPerFeatureProvider " + std::to_string(catchment_id);
176 uce.provider_bmi_var_name = output_name;
177 uce.provider_units = available_forcings_units[output_name];
178 uce.unconverted_values.push_back(value);
179 throw uce;
180 }
181 }

244 inline void check_forcing_vector_index_bounds()
245 {
246 std::stringstream ss;
247 //Check if forcing index is less than zero and if so, set to zero.
248 if (forcing_vector_index < 0)
249 {
250 forcing_vector_index = 0;
252 ss << "WARNING: Forcing vector index is less than zero. Therefore, setting index to zero." << std::endl;
253 LOG(ss.str(), LogLevel::SEVERE); ss.str("");
254 }
255
256 //Check if forcing index is greater than or equal to the size of the size of the time vector and if so, set to zero.
257 else if (forcing_vector_index >= time_epoch_vector.size())
258 {
259 forcing_vector_index = time_epoch_vector.size() - 1;
261 ss << "WARNING: Reached beyond the size of the forcing vector. Therefore, setting index to last value of the vector." << std::endl;
262 LOG(ss.str(), LogLevel::SEVERE); ss.str("");
263 }
264
265 return;
266 }

303 void read_csv(std::string file_name)
304 {
305 int time_col_index = 0;
306 //std::map<std::string, int> col_indices;
307 std::vector<std::vector<double>*> local_valvec_index = {};
308
309 //Call CSVReader constuctor
310 CSVReader reader(file_name);
311
312 //Get the data from CSV File
313 std::vector<std::vector<std::string> > data_list = reader.getData();
314
315 // Process the header (first) row..
316 int col_num = 0;
317 for (const auto& col_head : data_list[0]){
318 //std::cerr << s << std::endl;
319 if(col_head == "Time" || col_head == "time"){
320 time_col_index = col_num;
321 local_valvec_index.push_back(nullptr); // make sure the column indices line up!
322 } else {
323 std::string var_name = col_head;
324 std::string units = "";
325
326 boost::trim(var_name); // remove leading/trailing ws
327 const auto var_name_close = var_name.back();
328 if (var_name_close == ']' || var_name_close == ')') {
329 // found closing bracket/parenth
330
331 const bool is_bracket = var_name_close == ']';
332 const size_t var_name_open = is_bracket ? var_name.rfind('[') : var_name.rfind('(');
333 if (var_name_open != std::string::npos) {
334 // found matching opening bracket/parenth
335
336 units = var_name.substr(var_name_open + 1);
337 units.pop_back(); // remove closing bracket
338
339 var_name = var_name.substr(0, var_name_open);
340 boost::trim(var_name); // trim again in case of ws between name and units
341 }
342 }
343
344 LOG("CsvProvider has variable '" + var_name + "' with units '" + units + "'", LogLevel::DEBUG);
345
346 auto wkf = data_access::WellKnownFields.find(var_name);
347 if(wkf != data_access::WellKnownFields.end()){
348 units = units.empty() ? std::get<1>(wkf->second) : units;
349 auto wkf_name = std::get<0>(wkf->second);
350 LOG("CsvProvider has well-known name '" + wkf_name + "' for variable '" + var_name + "' with units '" + units + "'", LogLevel::DEBUG);
351 available_forcings.push_back(var_name); // Allow lookup by non-canonical name
352 available_forcings_units[var_name] = units; // Allow lookup of units by non-canonical name
353 var_name = wkf_name; // Use the CSDMS name from here on
354 }
355
356 forcing_vectors[var_name] = {};
357 local_valvec_index.push_back(&(forcing_vectors[var_name]));
358 available_forcings.push_back(var_name);
359 available_forcings_units[var_name] = units;
360 }
361 col_num++;
362 }
363
364 time_t current_row_date_time_epoch;
365 //Iterate through CSV starting on the second row
366 int i = 1;
367 for (i = 1; i < data_list.size(); i++)
368 {
369 //Row vector
370 std::vector<std::string>& vec = data_list[i];
371
372 struct tm current_row_date_time_utc = tm();
373 std::string time_str = vec[time_col_index];
374 //TODO: Support more time string formats? This is basically ISO8601 but not complete, support TZ?
375 strptime(time_str.c_str(), "%Y-%m-%d %H:%M:%S", &current_row_date_time_utc);
376
377 //Convert current row date-time UTC to epoch time
378 current_row_date_time_epoch = timegm(&current_row_date_time_utc);
379
380 //TODO: I am not sure this is a concern of this object. If forcing is retrieved that doesn't cover the
381 //needed time period, isn't that the requester's concern? (Methods exist to check this...)
382 //Ensure that forcing data covers the entire model period. Otherwise, throw an error.
383 if (i == 1 && start_date_time_epoch < current_row_date_time_epoch)
384 {
385 struct tm start_date_tm;
386 gmtime_r(&start_date_time_epoch, &start_date_tm);
387
388 char tm_buff[128];
389 strftime(tm_buff, 128, "%Y-%m-%d %H:%M:%S", &start_date_tm);
390 std::string throw_msg; throw_msg.assign("Error: Forcing data " + file_name + " begins after the model start time:" + std::string(tm_buff) + " < " + time_str);
391 LOG(throw_msg, LogLevel::WARNING);
392 throw std::out_of_range(throw_msg);
393 }
394
395
396 if (start_date_time_epoch <= current_row_date_time_epoch && current_row_date_time_epoch <= end_date_time_epoch)
397 {
398 time_epoch_vector.push_back(current_row_date_time_epoch);
399
400 int c = -1;
401 for (auto& s : vec){
402 c++;
403 if(c == time_col_index)
404 continue;
405 boost::algorithm::trim(s);
406 local_valvec_index[c]->push_back(boost::lexical_cast<double>(s)); // This is supposed to update the vector in the map...
407 }
408
409 }
410
411 }
412 if (i <= 1 || current_row_date_time_epoch < end_date_time_epoch)
413 {
415 std::stringstream ss;
416 ss << "WARNING: Forcing data ends before the model end time." << std::endl;
417 LOG(ss.str(), LogLevel::SEVERE); ss.str("");
418 //std::string throw_msg; throw_msg.assign("Error: Forcing data ends before the model end time.");
419 }
420 }
