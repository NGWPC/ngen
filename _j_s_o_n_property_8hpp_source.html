173 JSONProperty(std::string value_key, const boost::property_tree::ptree& property_tree):
174 key(std::move(value_key)) {
175
176 if(property_tree.empty() && property_tree.data().empty()) {
177 //Since property trees don't represent empty strings, list, or objects, they are all just `empty`
178 //so we can trap that case with `property_tree.data().empty()`, and the best we can do is make it an
179 //empty string
180 type = PropertyType::String;
181 //Also note that a boost::variant assigned to the static empty string "" like this
182 //`data = "";`
183 //will cause `data` to actually become a bool type, not the std::string.
184 //So use a default string to intialize the empty data
185 data = std::string();
186 }
187 else if (property_tree.empty() && !property_tree.data().empty()) {
188 // This is a terminal node and has a raw value
189
190 const std::string& value = property_tree.data();
191
192 if (value == "true" || value == "false") {
193 type = PropertyType::Boolean;
194 data = value == "true";
195 }
196 else {
197 //Try natural number first since double cant cast to int/long
198 long casted_long_data;
199 double casted_double_data;
200 if( boost::conversion::try_lexical_convert<long>(value, casted_long_data) ){
201 type = PropertyType::Natural;
202 data = casted_long_data;
203 }
204 else if( boost::conversion::try_lexical_convert<double>(value, casted_double_data) ){
205 //Try to cast to double/real next
206 type = PropertyType::Real;
207 data = casted_double_data;
208 }
209 else{
210 //At this point, we are left with string option
211 //string = value;
212 type = PropertyType::String;
213 data = std::move(value);
214 }
215 }
216 }
217 else {
218 // This isn't a terminal node, therefore represents an object or array
219 //TODO unit test these construction paths...
220 for (auto &property : property_tree) {
221 if (property.first.empty()) {
222 type = PropertyType::List;
223 value_list.push_back(JSONProperty(value_key, property.second));
224 data = List(& value_list );
225 }
226 else {
227 type = PropertyType::Object;
228 values.emplace(property.first, JSONProperty(property.first, property.second));
229 data = Object( & values );
230 }
231 }
232 }
233 }

313 JSONProperty(std::string value_key, std::string value):key(std::move(value_key)) {
314 if (value == "true" || value == "false") {
315 type = PropertyType::Boolean;
316 //boolean = value == "true";
317 data = value == "true";
318 }
319 else {
320 bool is_numeric = true;
321 bool is_real = true;
322 bool decimal_already_hit = false;
323
324 for(int character_index = 0; character_index < value.length(); character_index++) {
325 char character = value[character_index];
326
327 // If the first character is a '0' or isn't a digit, the whole value cannot be a number
328 if (character_index == 0 && character == '0') {
329 is_numeric = false;
330 }
331 else if (character != '.' && !std::isdigit(character)) {
332 // If this character isn't a decimal point and isn't a digit, the whole value cannot be a number
333 is_numeric = false;
334 is_real = false;
335 break;
336 }
337 else if (character == '.' && decimal_already_hit) {
338 // If this character is a decimal point, but we've already seen one, the whole value cannot be a number
339 is_real = false;
340 break;
341 }
342 else if (character == '.') {
343 // If a decimal point is seen, the whole value cannot be an integer
344 is_numeric = false;
345 decimal_already_hit = true;
346 }
347 }
348
349 // If the value can be represented as a whole number, we want to go with that
350 if (is_numeric) {
351 type = PropertyType::Natural;
352 //natural_number = std::stol(value);
353 data = std::stol(value);
354 }
355 else if (is_real) {
356 type = PropertyType::Real;
357 //real_number = std::stod(value);
358 data = std::stod(value);
359 }
360 else {
361 // Otherwise we'll store everything as a raw string
362 type = PropertyType::String;
363 //string = value;
364 data = value;
365 }
366 }
367 }

446 static std::string print_property(const geojson::JSONProperty& p, int tab = 0, bool newline = true) {
447 std::ostringstream output; // Use a string stream instead of cout
448 char end = newline ? '\n' : '\0';
449
450 output << std::setw(tab); // Maintain indentation
451
452 switch (p.get_type()) {
453 case geojson::PropertyType::String:
454 output << p.as_string() << end;
455 break;
456 case geojson::PropertyType::Real:
457 output << p.as_real_number() << end;
458 break;
459 case geojson::PropertyType::Natural:
460 output << p.as_natural_number() << end;
461 break;
462 case geojson::PropertyType::Boolean:
463 output << (p.as_boolean() ? "true" : "false") << end;
464 break;
465 case geojson::PropertyType::List:
466 output << std::setw(tab) << "[";
467 tab += 5;
468 for (const auto& lp : p.as_list()) {
469 //This is a little harder to align nicely without knowing
470 //the length of the property as a string first...so for now,
471 //just try to get a little bit in to make it easier to read
472 output << std::setw(tab);
473 output << print_property(lp, tab, false); // Recursively append
474 output << "," << end;
475 }
476 tab -= 5;
477 output << std::setw(tab) << " ]" << end;
478 break;
479 case geojson::PropertyType::Object:
480 output << std::setw(tab) << "{\n";
481 tab += 5;
482 for (const auto& pair : p.get_values()) {
483 output << std::setw(tab + pair.first.length()) << pair.first << " : ";
484 output << print_property(pair.second, tab, false); // Recursively append
485 output << ",\n";
486 }
487 tab -= 5;
488 output << std::setw(tab) << "}" << end;
489 break;
490 }
491
492 return output.str();
493 }
