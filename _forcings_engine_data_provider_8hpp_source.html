169 ForcingsEngineDataProvider(
170 const std::string& init_config,
171 std::size_t time_begin_seconds,
172 std::size_t time_end_seconds
173 )
174 : time_begin_(std::chrono::system_clock::from_time_t(time_begin_seconds))
175 , time_end_(std::chrono::system_clock::from_time_t(time_end_seconds))
176 {
177 std::stringstream ss;
178
179 // Log the constructor arguments
180 ss.str("");
181 ss << "Entering ForcingsEngineDataProvider constructor" << std::endl;
182 LOG(LogLevel::DEBUG, ss.str());
183
184 std::time_t start_t = static_cast<std::time_t>(time_begin_seconds);
185 std::time_t end_t = static_cast<std::time_t>(time_end_seconds);
186
187 ss.str("");
188 ss << " Start time: " << std::put_time(std::gmtime(&start_t), "%Y-%m-%d %H:%M:%S UTC")
189 << " (" << time_begin_seconds << ")" << std::endl;
190 LOG(LogLevel::INFO, ss.str());
191
192 ss.str("");
193 ss << " End time: " << std::put_time(std::gmtime(&end_t), "%Y-%m-%d %H:%M:%S UTC")
194 << " (" << time_end_seconds << ")" << std::endl;
195 LOG(LogLevel::INFO, ss.str());
196
197 // Attempt to retrieve a previously created BMI instance
198 bmi_ = storage_type::instances.get(init_config);
199
200 // If it doesn't exist, create it and assign it to the storage map
201 if (bmi_ != nullptr) {
202 ss.str("");
203 ss << "Reusing existing BMI instance for init_config file: " << init_config << std::endl;
204 LOG(LogLevel::DEBUG, ss.str());
205 } else {
206 // Log the creation of a new BMI instance
207 ss.str(""); ss << "Creating new BMI instance for init_config file: " << init_config << std::endl;
208 LOG(LogLevel::DEBUG, ss.str());
209
210 // Create the BMI instance
211 try {
212 bmi_ = std::make_shared<models::bmi::Bmi_Py_Adapter>(
213 "ForcingsEngine",
214 init_config,
215 forcings_engine_python_classpath,
216 /*has_fixed_time_step=*/true
217 );
218 } catch (const std::exception& ex) {
219 ss.str("");ss << "Failed to create Bmi_Py_Adapter: " << ex.what() << std::endl;
220 LOG(LogLevel::FATAL, ss.str());
221 throw;
222 }
223
224 // Store the new BMI instance
225 storage_type::instances.set(init_config, bmi_);
226 }
227
228 try {
229 // Now, initialize the BMI dependent instance members
230 // NOTE: using std::lround instead of static_cast will prevent potential UB
231 time_step_ = std::chrono::seconds{std::lround(bmi_->GetTimeStep())};
232 var_output_names_ = bmi_->GetOutputVarNames();
233 if (Logger::GetLogger()->GetLogLevel() == LogLevel::DEBUG) {
234 ss.str(""); ss << "BMI instance initialized successfully" << std::endl;
235 LOG(LogLevel::DEBUG, ss.str());
236 ss.str(""); ss << "Time step: " << time_step_.count() << " seconds" << std::endl;
237 LOG(LogLevel::DEBUG, ss.str());
238 ss.str(""); ss << "Available output variable names:" << std::endl;
239 LOG(LogLevel::DEBUG, ss.str());
240 for (const auto& var_name : var_output_names_) {
241 ss.str(""); ss << " - " << var_name << std::endl;
242 LOG(LogLevel::DEBUG, ss.str());
243 }
244 }
245 } catch (const std::exception& ex) {
246 ss.str(""); ss << "Error initializing BMI instance: " << ex.what() << std::endl;
247 LOG(LogLevel::FATAL, ss.str());
248 throw;
249 }
250
251 // Log successful constructor exit
252 ss.str(""); ss << "Exiting ForcingsEngineDataProvider constructor" << std::endl;
253 LOG(LogLevel::DEBUG, ss.str());
254
255 }

258 std::string ensure_variable(std::string name, const std::string& suffix = "_ELEMENT") const
259 {
260 // TODO: use get_available_var_names() once const
261 auto vars = boost::span<const std::string>{var_output_names_};
262
263 // Check if the variable exists as-is
264 if (std::find(vars.begin(), vars.end(), name) != vars.end()) {
265 return name;
266 }
267
268 // Check if the suffixed version exists
269 auto suffixed_name = std::move(name) + suffix;
270 if (std::find(vars.begin(), vars.end(), suffixed_name) != vars.end()) {
271 return suffixed_name;
272 }
273
274 // If neither exists, throw an exception
275 throw std::runtime_error{
276 "ForcingsEngineDataProvider: neither variable `"
277 + suffixed_name.substr(0, suffixed_name.size() - suffix.size())
278 + "` nor `" + suffixed_name + "` exist."
279 };
280 }
