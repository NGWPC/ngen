55 void read(geojson::GeoJSON fabric, utils::StreamHandler output_stream) {
56 std::stringstream ss;
57 ss.str(""); ss << "Entering Formulation_Manager::read()" << std::endl;
58 LOG(ss.str(), LogLevel::DEBUG);
59
60 //TODO seperate the parsing of configuration options like time
61 //and routing and other non feature specific tasks from this main function
62 //which has to iterate the entire hydrofabric.
63 auto possible_global_config = tree.get_child_optional("global");
64
65 if (possible_global_config) {
66 global_config = realization::config::Config(*possible_global_config);
67 }
68
69 // Log simulation time configuration
70 auto possible_simulation_time = tree.get_child_optional("time");
71
72 if (!possible_simulation_time) {
73 std::string throw_msg; throw_msg.assign("ERROR: No simulation time period defined.");
74 LOG(throw_msg, LogLevel::WARNING);
75 throw std::runtime_error(throw_msg);
76 }
77
78 config::Time time = config::Time(*possible_simulation_time);
79 auto simulation_time_config = time.make_params();
80
81 // Initialize the Simulation_Time object
82 this->Simulation_Time_Object = std::make_shared<Simulation_Time>(simulation_time_config);
83
84 // Log layer descriptions
85 // try to get the json node
86 auto layers_json_array = tree.get_child_optional("layers");
87 //Create the default surface layer
88 config::Layer layer;
89 // layer description struct
90 ngen::LayerDescription layer_desc;
91 layer_desc = layer.get_descriptor();
92 // add the default surface layer to storage
93 layer_storage.put_layer(layer_desc, layer_desc.id);
94
95 if (layers_json_array) {
96
97 for (std::pair<std::string, boost::property_tree::ptree> layer_config : *layers_json_array)
98 {
99 layer = config::Layer(layer_config.second);
100 layer_desc = layer.get_descriptor();
101
102 // add the layer to storage
103 layer_storage.put_layer(layer_desc, layer_desc.id);
104 ss.str(""); ss << "Layer added: ID = " << layer_desc.id << ", Name = " << layer_desc.name << std::endl;
105 LOG(ss.str(), LogLevel::DEBUG);
106
107 if (layer.has_formulation() && layer.get_domain() == "catchments") {
108 double c_value = UnitsHelper::get_converted_value(layer_desc.time_step_units, layer_desc.time_step, "s");
109 // make a new simulation time object with a different output interval
110 Simulation_Time sim_time(*Simulation_Time_Object, c_value);
111 domain_formulations.emplace(
112 layer_desc.id,
113 construct_formulation_from_config(
114 simulation_time_config,
115 "layer-" + std::to_string(layer_desc.id),
116 layer.formulation,
117 output_stream
118 )
119 );
120 domain_formulations.at(layer_desc.id)->set_output_stream(get_output_root() + layer_desc.name + "_layer_"+std::to_string(layer_desc.id) + ".csv");
121 }
122 //TODO for each layer, create deferred providers for use by other layers
123 //VERY SIMILAR TO NESTED MODULE INIT
124 }
125 }
126
127 //TODO use the set of layer providers as input for catchments to lookup from
128
129 // Read routing configurations from configuration file
130 auto possible_routing_configs = tree.get_child_optional("routing");
131
132 if (possible_routing_configs) {
133 // Since it is possible to build NGEN without routing support, if we see it in the config
134 // but it isn't enabled in the build, we should at least put up a warning
135 #if NGEN_WITH_ROUTING
136 this->routing_config = (config::Routing(*possible_routing_configs)).params;
137 using_routing = true;
138 #else
139 using_routing = false;
140 ss.str("");
141 ss <<"Formulation Manager found routing configuration"
142 << ", but routing support isn't enabled. No routing will occur." << std::endl;
143 LOG(ss.str(), LogLevel::WARNING);
144 #endif // NGEN_WITH_ROUTING
145 }
146
147 std::unordered_map<std::string, realization::config::Formulation> formulation_groups;
148 auto possible_formulation_groups = tree.get_child_optional("formulation_groups");
149 if (possible_formulation_groups) {
150 for (std::pair<std::string, boost::property_tree::ptree> formulation_config : *possible_formulation_groups) {
151 std::cout << formulation_config.first.c_str() << std::endl;
152 realization::config::Formulation formulation(formulation_config.second.get_child(".")); // "." for first element in list
153 formulation_groups[formulation_config.first] = formulation;
154 }
155 }
156
157 std::unordered_map<std::string, realization::config::Forcing> forcing_groups;
158 auto possible_forcing_groups = tree.get_child_optional("forcing_groups");
159 if (possible_forcing_groups) {
160 for (std::pair<std::string, boost::property_tree::ptree> forcing_config : *possible_forcing_groups) {
161 realization::config::Forcing forcing(forcing_config.second);
162 forcing_groups[forcing_config.first] = forcing;
163 }
164 }
165
169 auto possible_catchment_configs = tree.get_child_optional("catchments");
170
171 if (possible_catchment_configs) {
172 for (std::pair<std::string, boost::property_tree::ptree> catchment_config : *possible_catchment_configs) {
173 ss.str(""); ss << "Processing catchment: " << catchment_config.first << std::endl;
174 LOG(ss.str(), LogLevel::DEBUG);
175
176 int catchment_index = fabric->find(catchment_config.first);
177 if (catchment_index == -1) {
178 #ifndef NGEN_QUIET
179 ss.str("");
180 ss << "Formulation_Manager::read: Cannot create formulation for catchment "
181 << catchment_config.first
182 << " that isn't identified in the hydrofabric or requested subset" << std::endl;
183 LOG(ss.str(), LogLevel::WARNING);
184 #endif
185 continue;
186 }
187 realization::config::Config catchment_formulation(catchment_config.second);
188
189 if (!catchment_formulation.has_formulation()) {
190 std::string throw_msg;
191 throw_msg.assign("ERROR: No formulations defined for " + catchment_config.first + ".");
192 LOG(throw_msg, LogLevel::WARNING);
193 throw std::runtime_error(throw_msg);
194 }
195
196 // Parse catchment-specific model_params
197 auto catchment_feature = fabric->get_feature(catchment_index);
198 ss.str(""); ss << "Linking external properties for catchment: " << catchment_config.first << std::endl;
199 LOG(ss.str(), LogLevel::DEBUG);
200 catchment_formulation.formulation.link_external(catchment_feature);
201
202 this->add_formulation(
203 this->construct_formulation_from_config(
204 simulation_time_config,
205 catchment_config.first,
206 catchment_formulation,
207 output_stream
208 )
209 );
210 ss.str(""); ss << "Formulation constructed for catchment: " << catchment_config.first << std::endl;
211 LOG(ss.str(), LogLevel::DEBUG);
212 }
213 }
214
215 // Process any catchments not explicitly defined in the realization file
216 for (geojson::Feature location : *fabric) {
217 if (not this->contains(location->get_id())) {
218 ss.str(""); ss << "Creating missing formulation for location: " << location->get_id() << std::endl;
219 LOG(ss.str(), LogLevel::DEBUG);
220 std::shared_ptr<Catchment_Formulation> missing_formulation = this->construct_missing_formulation(
221 location, output_stream, simulation_time_config);
222 this->add_formulation(missing_formulation);
223// ss.str(""); ss << "Missing formulation created for location: " << location->get_id() << std::endl;
224// LOG(ss.str(), LogLevel::DEBUG);
225 }
226 }
227 }

297 // The calls in these loops are staticly dispatched to
298 // Catchment_Formulation::finalize(). That does not
299 // inherit from DataProvider, with its virtual member
300 // function of the same name.

346 std::string get_output_root() const {
347 const auto output_root = this->tree.get_optional<std::string>("output_root");
348 if (output_root != boost::none && *output_root != "") {
349 // Check if the path ends with a trailing slash,
350 // otherwise add it.
351 std::string str = output_root->back() == '/'
352 ? *output_root
353 : *output_root + "/";
354
355 const char* dir = str.c_str();
356
357 //use C++ system function to check if there is a dir match that defined in realization
358 struct stat sb;
359 if (stat(dir, &sb) == 0 && S_ISDIR(sb.st_mode)) {
360 return str;
361 } else {
362 errno = 0;
363 int result = mkdir(dir, 0755);
364 if (result == 0)
365 return str;
366 else
367 throw std::runtime_error("failed to create directory '" + str + "': " + std::strerror(errno));
368 }
369 }
370
371 //for case where there is no output_root in the realization file
372 return "./";
373 }

388 std::shared_ptr<Catchment_Formulation> construct_formulation_from_config(
389 simulation_time_params &simulation_time_config,
390 std::string identifier,
391 const realization::config::Config& catchment_formulation,
392 utils::StreamHandler output_stream
393 ) {
394 std::stringstream ss;
395 ss.str(""); ss << "Entering construct_formulation_from_config for identifier: " << identifier << std::endl;
396 LOG(ss.str(), LogLevel::DEBUG);
397
398 // Check if the formulation exists
399 if (!formulation_exists(catchment_formulation.formulation.type)) {
400 std::string throw_msg;
401 throw_msg.assign("Catchment " + identifier + " failed initialization: '" +
402 catchment_formulation.formulation.type + "' is not a valid formulation. Options are: " +
403 valid_formulation_keys());
404 LOG(throw_msg, LogLevel::WARNING);
405 throw std::runtime_error(throw_msg);
406 }
407
408 // Check for missing forcing parameters
409 ss.str(""); ss << "Checking forcing parameters for identifier: " << identifier << std::endl;
410 LOG(ss.str(), LogLevel::DEBUG);
411 if (catchment_formulation.forcing.parameters.empty()) {
412 std::string throw_msg;
413 throw_msg.assign("No forcing definition was found for " + identifier);
414 LOG(throw_msg, LogLevel::FATAL);
415 throw std::runtime_error(throw_msg);
416 }
417
418 // Check for missing path
419 std::vector<std::string> missing_parameters;
420 if (!catchment_formulation.forcing.has_key("path")) {
421 ss.str(""); ss << "Missing path parameter for identifier: " << identifier << std::endl;
422 LOG(ss.str(), LogLevel::DEBUG);
423 missing_parameters.push_back("path");
424 }
425
426 // Log missing parameters, if any
427 if (!missing_parameters.empty()) {
428 std::string message = "A forcing configuration cannot be created for '" + identifier + "'; the following parameters are missing: ";
429 for (size_t i = 0; i < missing_parameters.size(); ++i) {
430 message += missing_parameters[i];
431 if (i < missing_parameters.size() - 1) {
432 message += ", ";
433 }
434 }
435
436 std::string throw_msg;
437 throw_msg.assign(message);
438 LOG(throw_msg, LogLevel::WARNING);
439 throw std::runtime_error(throw_msg);
440 }
441
442 // Extract forcing parameters
443 forcing_params forcing_config = this->get_forcing_params(catchment_formulation.forcing.parameters, identifier, simulation_time_config);
444 ss.str("");
445 ss << "Forcing parameters extracted for identifier: " << identifier << std::endl;
446 ss << " Forcing path: " << forcing_config.path << std::endl;
447 ss << " Forcing provider: " << forcing_config.provider << std::endl;
448 ss << " Forcing init_config: " << forcing_config.init_config << std::endl;
449 LOG(ss.str(), LogLevel::DEBUG);
450
451 std::time_t start_t = static_cast<std::time_t>(forcing_config.simulation_start_t);
452 std::time_t end_t = static_cast<std::time_t>(forcing_config.simulation_end_t);
453
454 ss.str("");
455 ss << " Simulation start time: " << std::put_time(std::gmtime(&start_t), "%Y-%m-%d %H:%M:%S UTC")
456 << " (" << forcing_config.simulation_start_t << ")" << std::endl;
457 ss << " Simulation end time: " << std::put_time(std::gmtime(&end_t), "%Y-%m-%d %H:%M:%S UTC")
458 << " (" << forcing_config.simulation_end_t << ")" << std::endl;
459 LOG(ss.str(), LogLevel::DEBUG);
460
461 // Construct formulation
462 ss.str(""); ss << "Constructing formulation for type: " << catchment_formulation.formulation.type << std::endl;
463 LOG(ss.str(), LogLevel::DEBUG);
464 std::shared_ptr<Catchment_Formulation> constructed_formulation = construct_formulation(
465 catchment_formulation.formulation.type, identifier, forcing_config, output_stream
466 );
467 ss.str(""); ss << "Formulation constructed successfully for identifier: " << identifier << std::endl;
468 LOG(ss.str(), LogLevel::DEBUG);
469
470 // Create formulation instance
471 ss.str(""); ss << "Calling create_formulation for identifier: " << identifier << std::endl;
472 LOG(ss.str(), LogLevel::DEBUG);
473 constructed_formulation->create_formulation(catchment_formulation.formulation.parameters);
474 ss.str(""); ss << "Formulation creation completed for identifier: " << identifier << std::endl;
475 LOG(ss.str(), LogLevel::DEBUG);
476
477 return constructed_formulation;
478 }

603 forcing_params get_forcing_params(const geojson::PropertyMap &forcing_prop_map, std::string identifier, simulation_time_params &simulation_time_config) {
604 std::stringstream ss;
605 ss.str(""); ss << "Entering get_forcing_params for identifier: " << identifier << std::endl;
606 LOG(ss.str(), LogLevel::DEBUG);
607
608 // Extract the required 'path' parameter
609 std::string path = "";
610 if (forcing_prop_map.count("path") != 0) {
611 path = forcing_prop_map.at("path").as_string();
612 ss.str(""); ss << " Forcing path: " << path << std::endl;
613 LOG(ss.str(), LogLevel::DEBUG);
614 }
615
616 // Extract the required 'provider' parameter
617 std::string provider;
618 if (forcing_prop_map.count("provider") != 0) {
619 provider = forcing_prop_map.at("provider").as_string();
620 ss.str(""); ss << " Forcing provider: " << provider << std::endl;
621 LOG(ss.str(), LogLevel::DEBUG);
622 }
623
624 // Extract the optional 'init_config' parameter from 'params'
625 std::string init_config = "";
626 if (forcing_prop_map.count("params") != 0) {
627 const geojson::JSONProperty& params_property = forcing_prop_map.at("params");
628 if (params_property.get_type() == geojson::PropertyType::Object) {
629 const geojson::PropertyMap& params_map = params_property.get_values();
630 if (params_map.count("init_config") != 0) {
631 init_config = params_map.at("init_config").as_string();
632 ss.str(""); ss << " Forcing init_config: " << init_config << std::endl;
633 LOG(ss.str(), LogLevel::DEBUG);
634 }
635 } else {
636 std::cout << "[WARNING] 'params' is not an object for identifier: " << identifier << std::endl;
637 }
638 }
639
640 // If no file pattern is present, return the parameters directly
641 if (forcing_prop_map.count("file_pattern") == 0) {
642 return forcing_params(
643 path,
644 provider,
645 simulation_time_config.start_time,
646 simulation_time_config.end_time,
647 init_config
648 );
649 }
650
651 // Ensure the 'path' is set for pattern matching
652 if (path.empty()) {
653 std::string throw_msg = "Error with NGEN config - 'path' in forcing params must be set to a "
654 "non-empty parent directory path when 'file_pattern' is used.";
655 LOG(throw_msg, LogLevel::WARNING);
656 throw std::runtime_error(throw_msg);
657 }
658
659 // Append a trailing slash to the path if not already present
660 if (path.compare(path.size() - 1, 1, "/") != 0) {
661 path += "/";
662 }
663
664 // Extract and process the file pattern
665 std::string filepattern = forcing_prop_map.at("file_pattern").as_string();
666 int id_index = filepattern.find("{{id}}");
667
668 // Replace {{id}} if present
669 if (id_index != std::string::npos) {
670 filepattern = filepattern.replace(id_index, sizeof("{{id}}") - 1, identifier);
671 }
672
673 // Compile the file pattern as a regex
674 std::regex pattern(filepattern);
675
676 // A stream providing the functions necessary for evaluating a directory:
677 // https://www.gnu.org/software/libc/manual/html_node/Opening-a-Directory.html#Opening-a-Directory
678 DIR *directory = nullptr;
679
680 // structure representing the member of a directory: https://www.gnu.org/software/libc/manual/html_node/Directory-Entries.html
681 struct dirent *entry = nullptr;
682
683 // Attempt to open the directory for evaluation
684 directory = opendir(path.c_str());
685 // Allow for a few retries in certain failure situations
686 size_t attemptCount = 0;
687 std::string errMsg;
688
689 // Retry on certain error codes
690 while (directory == nullptr && attemptCount++ < 5) {
691 // For several error codes, we should break immediately and not retry
692 if (errno == ENOENT) {
693 errMsg = "No such file or directory.";
694 break;
695 }
696 if (errno == ENXIO) {
697 errMsg = "No such device or address.";
698 break;
699 }
700 if (errno == EACCES) {
701 errMsg = "Permission denied.";
702 break;
703 }
704 if (errno == EPERM) {
705 errMsg = "Operation not permitted.";
706 break;
707 }
708 if (errno == ENOTDIR) {
709 errMsg = "File at provided path is not a directory.";
710 break;
711 }
712 if (errno == EMFILE) {
713 errMsg = "The current process has too many open files.";
714 break;
715 }
716 if (errno == ENFILE) {
717 errMsg = "The system has too many open files.";
718 break;
719 }
720
721 // Sleep before retrying to avoid a tight loop
722 sleep(2);
723 directory = opendir(path.c_str());
724 errMsg = "Received system error number " + std::to_string(errno);
725 }
726
727 // Iterate over directory entries
728 if (directory != nullptr) {
729 while ((entry = readdir(directory))) {
730 if (std::regex_match(entry->d_name, pattern)) {
731 // Check for regular files and symlinks
732 #ifdef _DIRENT_HAVE_D_TYPE
733 if (entry->d_type == DT_REG || entry->d_type == DT_LNK) {
734 closedir(directory);
735 return forcing_params(
736 path + entry->d_name,
737 provider,
738 simulation_time_config.start_time,
739 simulation_time_config.end_time,
740 init_config
741 );
742 }
743 else if (entry->d_type == DT_UNKNOWN) {
744 #endif
745 // Use stat for systems that don't set d_type
746 struct stat st;
747 if (stat((path + entry->d_name).c_str(), &st) != 0) {
748 std::string throw_msg = "Could not stat file " + path + entry->d_name;
749 LOG(throw_msg, LogLevel::WARNING);
750 throw std::runtime_error(throw_msg);
751 }
752
753 if (S_ISREG(st.st_mode)) {
754 closedir(directory);
755 return forcing_params(
756 path + entry->d_name,
757 provider,
758 simulation_time_config.start_time,
759 simulation_time_config.end_time,
760 init_config
761 );
762 }
763
764 // Log a warning if the entry is not a regular file
765 std::string throw_msg = "Forcing data in path " + path + entry->d_name + " is not a file";
766 LOG(throw_msg, LogLevel::WARNING);
767 throw std::runtime_error(throw_msg);
768 #ifdef _DIRENT_HAVE_D_TYPE
769 }
770 #endif
771 }
772 }
773 closedir(directory);
774 } else {
775 // The directory wasn't found or otherwise couldn't be opened; forcing data cannot be retrieved
776 std::string throw_msg = "Error opening forcing data dir '" + path + "' after " + std::to_string(attemptCount) + " attempts: " + errMsg;
777 LOG(throw_msg, LogLevel::WARNING);
778 throw std::runtime_error(throw_msg);
779 }
780
781 // If no match was found, throw an error
782 std::string throw_msg = "Forcing data could not be found for '" + identifier + "'";
783 LOG(throw_msg, LogLevel::WARNING);
784 throw std::runtime_error(throw_msg);
785 }

794 void parse_external_model_params(boost::property_tree::ptree& model_params, const geojson::Feature catchment_feature) {
795 std::stringstream ss;
796 boost::property_tree::ptree attr {};
797 for (decltype(auto) param : model_params) {
798 if (param.second.count("source") == 0) {
799 attr.put_child(param.first, param.second);
800 continue;
801 }
802
803 decltype(auto) param_source = param.second.get_child("source");
804 decltype(auto) param_source_name = param_source.get_value<std::string>();
805 if (param_source_name != "hydrofabric") {
806 // temporary until the logic for alternative sources is designed
807 throw std::logic_error("ERROR: 'model_params' source `" + param_source_name + "` not currently supported. Only `hydrofabric` is supported.");
808 }
809
810 decltype(auto) param_name = param.second.find("from") == param.second.not_found()
811 ? param.first
812 : param.second.get_child("from").get_value<std::string>();
813
814 if (catchment_feature->has_property(param_name)) {
815 auto catchment_attribute = catchment_feature->get_property(param_name);
816 switch (catchment_attribute.get_type()) {
817 case geojson::PropertyType::Natural:
818 attr.put(param.first, catchment_attribute.as_natural_number());
819 break;
820 case geojson::PropertyType::Boolean:
821 attr.put(param.first, catchment_attribute.as_boolean());
822 break;
823 case geojson::PropertyType::Real:
824 attr.put(param.first, catchment_attribute.as_real_number());
825 break;
826 case geojson::PropertyType::String:
827 attr.put(param.first, catchment_attribute.as_string());
828 break;
829
830 case geojson::PropertyType::List:
831 case geojson::PropertyType::Object:
832 default:
833 ss.str("");
834 ss << "property type " << static_cast<int>(catchment_attribute.get_type()) << " not allowed as model parameter. "
835 << "Must be one of: Natural (int), Real (double), Boolean, or String" << '\n';
836 LOG(ss.str(), LogLevel::WARNING); ss.str("");
837 break;
838 }
839 } else {
840 ss.str("");
841 ss << " Failed to parse external parameter: catchment `"
842 << catchment_feature->get_id()
843 << "` does not contain the property `"
844 << param_name << "`\n";
845 LOG(ss.str(), LogLevel::WARNING);
846 }
847 }
848
849 model_params.swap(attr);
850 }

859 void parse_external_model_params(geojson::PropertyMap& model_params, const geojson::Feature catchment_feature) {
860 std::stringstream ss;
861 geojson::PropertyMap attr {};
862 for (decltype(auto) param : model_params) {
863 // Check for type to short-circuit. If param.second is not an object, `.has_key()` will throw
864 if (param.second.get_type() != geojson::PropertyType::Object || !param.second.has_key("source")) {
865 attr.emplace(param.first, param.second);
866 continue;
867 }
868
869 decltype(auto) param_source = param.second.at("source");
870 decltype(auto) param_source_name = param_source.as_string();
871 if (param_source_name != "hydrofabric") {
872 // TODO: temporary until the logic for alternative sources is designed
873 throw std::logic_error("ERROR: 'model_params' source `" + param_source_name + "` not currently supported. Only `hydrofabric` is supported.");
874 }
875
876 // Property name in the feature properties is either
877 // the value of key "from", or has the same name as
878 // the expected model parameter key
879 decltype(auto) param_name = param.second.has_key("from")
880 ? param.second.at("from").as_string()
881 : param.first;
882
883 if (catchment_feature->has_property(param_name)) {
884 auto catchment_attribute = catchment_feature->get_property(param_name);
885
886 // Use param.first in the `.emplace` calls instead of param_name since
887 // the expected name is given by the key of the model_params values.
888 switch (catchment_attribute.get_type()) {
889 case geojson::PropertyType::Natural:
890 attr.emplace(param.first, geojson::JSONProperty(param.first, catchment_attribute.as_natural_number()));
891 break;
892 case geojson::PropertyType::Boolean:
893 attr.emplace(param.first, geojson::JSONProperty(param.first, catchment_attribute.as_boolean()));
894 break;
895 case geojson::PropertyType::Real:
896 attr.emplace(param.first, geojson::JSONProperty(param.first, catchment_attribute.as_real_number()));
897 break;
898 case geojson::PropertyType::String:
899 attr.emplace(param.first, geojson::JSONProperty(param.first, catchment_attribute.as_string()));
900 break;
901
902 case geojson::PropertyType::List:
903 case geojson::PropertyType::Object:
904 default:
905 // TODO: Should list/object values be passed to model parameters?
906 // Typically, feature properties *should* be scalars.
907 ss.str(""); ss << "property type " << static_cast<int>(catchment_attribute.get_type()) << " not allowed as model parameter. "
908 << "Must be one of: Natural (int), Real (double), Boolean, or String" << '\n';
909 LOG(ss.str(), LogLevel::WARNING);
910 break;
911 }
912 }
913 }
914
915 model_params.swap(attr);
916 }
