232 const std::string get_analogous_cxx_type(const std::string &py_type_name, const size_t item_size) override {
233 /*
234 * Note that an implementation using a "switch" statement would be problematic. It could be done by
235 * rewriting to separate the integer and non-integer type, then having cases based on size. However,
236 * this might cause trouble on certain systems, since (depending on the particular sizes of types) that
237 * could produce duplicate "case" values.
238 */
239 //TODO: include other numpy type strings, https://numpy.org/doc/stable/user/basics.types.html
240 if ( (py_type_name == "int" || py_type_name == "int16") && item_size == sizeof(short)) {
241 return "short";
242 } else if ( (py_type_name == "int" || py_type_name == "int32" )&& item_size == sizeof(int)) {
243 return "int";
244 } else if (py_type_name == "int" && item_size == sizeof(long)) {
245 return "long";
246 } else if ( (py_type_name == "int" || py_type_name == "int64") && item_size == sizeof(long long)) {
247 return "long long";
248 } else if (py_type_name == "longlong" && item_size == sizeof(long long)) {
249 return "long long"; //numpy type
250 } else if ( (py_type_name == "float" || py_type_name == "float32" || py_type_name == "np.float32" ||
251 py_type_name == "numpy.float32" || py_type_name == "np.single" || py_type_name == "numpy.single") && item_size == sizeof(float)) {
252 return "float";
253 } else if ((py_type_name == "float" || py_type_name == "float64" || py_type_name == "np.float64" ||
254 py_type_name == "numpy.float64") && item_size == sizeof(double)) {
255 return "double";
256 } else if (py_type_name == "float" && item_size == sizeof(long double)) {
257 return "long double";
258 } else {
259 std::string throw_msg; throw_msg.assign(
260 "(Bmi_Py_Adapter) Failed determining analogous C++ type for Python model '" + py_type_name +
261 "' type with size " + std::to_string(item_size) + " bytes.");
262 LOG(throw_msg, LogLevel::WARNING);
263 throw std::runtime_error(throw_msg);
264 }
265 }

360 void get_and_copy_grid_array(const char* grid_func_name, const int grid, T* dest, int dest_length,
361 const char* np_dtype)
362 {
363 //This is required here because grid info can be a non dimensional `np.zeros( () )`

365 if( dest_length == 0 ){
366 np_array = np.attr("zeros")(1, "dtype"_a = np_dtype);
367 } else{
368 np_array = np.attr("zeros")(dest_length, "dtype"_a = np_dtype);
369 }
370 bmi_model->attr(grid_func_name)(grid, np_array);
371 auto np_array_direct = np_array.template unchecked<1>();
372 if( dest_length == 0 ){
373 dest[0] = np_array_direct(0);
374 }
375 else{
376 for (int i = 0; i < dest_length; ++i)
377 dest[i] = np_array_direct(i);

435 void get_value_at_indices(const std::string& name, void *dest, int *inds, int count, bool is_all_indices) {
436 std::string val_type = GetVarType(name);
437 size_t val_item_size = (size_t)GetVarItemsize(name);
438 std::vector<std::string> in_v = GetInputVarNames();
439
440 // The available types and how they are handled here should match what is in SetValueAtIndices
441 if (val_type == "int" && val_item_size == sizeof(short))
442 get_via_numpy_array<short>(name, dest, inds, count, val_item_size, is_all_indices);
443 else if (val_type == "int" && val_item_size == sizeof(int))
444 get_via_numpy_array<int>(name, dest, inds, count, val_item_size, is_all_indices);
445 else if (val_type == "int" && val_item_size == sizeof(long))
446 get_via_numpy_array<long>(name, dest, inds, count, val_item_size, is_all_indices);
447 else if (val_type == "int" && val_item_size == sizeof(long long))
448 get_via_numpy_array<long long>(name, dest, inds, count, val_item_size, is_all_indices);
449 else if (val_type == "float" || val_type == "float16" || val_type == "float32" || val_type == "float64") {
450 if (val_item_size == sizeof(float))
451 get_via_numpy_array<float>(name, dest, inds, count, val_item_size, is_all_indices);
452 else if (val_item_size == sizeof(double))
453 get_via_numpy_array<double>(name, dest, inds, count, val_item_size, is_all_indices);
454 else if (val_item_size == sizeof(long double))
455 get_via_numpy_array<long double>(name, dest, inds, count, val_item_size, is_all_indices);
456 }
457 else {
458 std::string throw_msg; throw_msg.assign(
459 "(Bmi_Py_Adapter) Failed attempt to GET values of BMI variable '" + name + "' from '" +
460 model_name + "' model: model advertises unsupported combination of type (" + val_type +
461 ") and size (" + std::to_string(val_item_size) + ").");
462 LOG(throw_msg, LogLevel::WARNING);
463 throw std::runtime_error(throw_msg);

488 template <typename T>
489 py::array_t<T> get_via_numpy_array(const std::string& name, void *dest, const int *indices, int item_count,
490 size_t item_size, bool is_all_indices)
491 {
492 std::string val_type = GetVarType(name);
493 py::array_t<T, py::array::c_style> dest_array
494 = np.attr("zeros")(item_count, "dtype"_a = val_type, "order"_a = "C");
495 if (is_all_indices) {
496 bmi_model->attr("get_value")(name, dest_array);
497 }
498 else {
499 py::array_t<int, py::array::c_style> indices_np_array
500 = np.attr("zeros")(item_count, "dtype"_a = "int32", "order"_a = "C");
501 auto indices_mut_direct = indices_np_array.mutable_unchecked<1>();
502 for (py::size_t i = 0; i < (py::size_t) item_count; ++i)
503 indices_mut_direct(i) = indices[i];
504 /* Leaving this here for now in case mutable_unchecked method doesn't behave as expected in testing
505 py::buffer_info buffer_info = indices_np_array.request();
506 int *indices_np_arr_ptr = (int*)buffer_info.ptr;
507 for (int i = 0; i < item_count; ++i)
508 indices_np_arr_ptr[i] = indices[i];
509 */
510 bmi_model->attr("get_value_at_indices")(name, dest_array, indices_np_array);
511 }
512
513 auto direct_access = dest_array.template unchecked<1>();
514 for (py::size_t i = 0; i < (py::size_t) item_count; ++i)
515 ((T *) dest)[i] = direct_access(i);
516
517 return dest_array;
518 }

533 void SetValue(std::string name, void *src) override {
534 int itemSize = GetVarItemsize(name);
535 std::string py_type = GetVarType(name);
536 std::string cxx_type = get_analogous_cxx_type(py_type, (size_t) itemSize);
537
538 if (cxx_type == "short") {
539 set_value<short>(name, (short *) src);
540 } else if (cxx_type == "int") {
541 set_value<int>(name, (int *) src);
542 } else if (cxx_type == "long") {
543 set_value<long>(name, (long *) src);
544 } else if (cxx_type == "long long") {
545 //FIXME this gets dicey -- if a python numpy array is of type np.int64 (long long),
546 //but a c++ int* is passed to this function as src, it will fail in undefined ways...
547 //the template type overload may be perferred for doing SetValue from framework components
548 //such as forcing providers...
549 set_value<long long>(name, (long long *) src);
550 } else if (cxx_type == "float") {
551 set_value<float>(name, (float *) src);
552 } else if (cxx_type == "double") {
553 set_value<double>(name, (double *) src);
554 } else if (cxx_type == "long double") {
555 set_value<long double>(name, (long double *) src);
556 } else {
557 std::string throw_msg; throw_msg.assign("Bmi_Py_Adapter cannot set values for variable '" + name +
558 "' that has unrecognized C++ type '" + cxx_type + "'");
559 LOG(throw_msg, LogLevel::WARNING);
560 throw std::runtime_error(throw_msg);
561 }

572 template <typename T>
573 void set_value(const std::string &name, std::vector<T> src) {
574 int nbytes = GetVarNbytes(name);
575 int itemSize = GetVarItemsize(name);
576 int length = nbytes / itemSize;
577
578 if (length != src.size()) {
579 std::string throw_msg; throw_msg.assign(
580 "Bmi_Py_Adapter mismatch of lengths setting variable array (" + std::to_string(length) +
581 " expected but " + std::to_string(src.size()) + " received)");

583 throw std::runtime_error(throw_msg);
584 }
585
586 py::array_t<T> model_var_array = bmi_model->attr("get_value_ptr")(name);
587 auto mutable_unchecked_proxy = model_var_array.template mutable_unchecked<1>();
588 for (size_t i = 0; i < length; ++i) {
589 mutable_unchecked_proxy(i) = src[i];
590 }

700 std::cerr << "Python error during BMI construction:\n" << e.what() << std::endl;

702 }

704 std::string err = "Failed in construct_and_init_backing_model_for_py_adapter: ";
705 err += e.what();
706 std::cout << err << std::endl;
707 throw std::runtime_error(err);
708 }
709 // Record the exception message before re-throwing to handle subsequent function calls properly
710 // TODO: handle exceptions in better detail, without losing type information
711 catch (std::exception& e) {

715 init_exception_msg = "Unknown Python model initialization exception.";
716 }
717 // This message is lost and often contains valuable info. Either need to break up and catch
718 // other possible exceptions, wrap all these in a custom exception, or at the very least, print
719 // the original message before it gets lost in this re-throw.
720 std::stringstream ss;
721 ss << init_exception_msg << std::endl;
722 std::cout << "Caught generic exception during BMI setup: " << init_exception_msg << std::endl;
723 LOG(ss.str(), LogLevel::WARNING); ss.str("");
724 throw e;
725 }

768 inline void separate_package_and_simple_name() {
769 if (!model_initialized) {
770 std::vector<std::string> split_name;
771 std::string delimiter = ".";
772 std::string name_string = bmi_type_py_full_name;
773
774 size_t pos = 0;
775 std::string token;
776 while ((pos = name_string.find(delimiter)) != std::string::npos) {
777 token = name_string.substr(0, pos);
778 split_name.emplace_back(token);
779 name_string.erase(0, pos + delimiter.length());
780 }
